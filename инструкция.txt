Мы будем использовать архитектуру Server-Authoritative (сервер — главный источник истины):
Клиент отвечает только за отрисовку (HTML/CSS), анимации, звуки и отправку намерений игрока ("хочу атаковать").
Сервер (Node.js + Socket.io) хранит состояние, проверяет правила, рассчитывает урон, управляет ИИ противника и отправляет клиенту обновленное состояние и события для анимации.

Сервер хранит состояние игрока, генерирует магазин, обрабатывает ходы и управляет ИИ.

const path = require('path');

const MAX_MANA = 10;

// Хранилище сессий игроков (в памяти, для продакшена нужна БД)
const players = new Map();

function generateId() { return Math.random().toString(36).substr(2, 9); }
function shuffle(array) { return [...array].sort(() => Math.random() - 0.5); }
function getCard(id) { return ALL_CARDS.find(c => c.id === id); }

io.on('connection', (socket) => {
  console.log('Игрок подключился:', socket.id);

  // Инициализация нового или загрузка существующего игрока
  if (!players.has(socket.id)) {
    players.set(socket.id, {
      gold: 100,
      collection: ['mage_01', 'tank_01', 'assa_01'],
      deck: ['mage_01', 'tank_01', 'assa_01'],
      shop: [],
      battle: null
    });
    generateShop(socket.id);
  }

  const getState = () => players.get(socket.id);

  // Отправка начального состояния
  socket.emit('init', getState());

  // ═══ МАГАЗИН ═══
  socket.on('buyCard', (cardId) => {
    const state = getState();
    const card = getCard(cardId);
    if (card && state.gold >= card.price && !state.collection.includes(cardId)) {
      state.gold -= card.price;
      state.collection.push(cardId);
      generateShop(socket.id); // Обновить магазин
      socket.emit('stateUpdate', state);
      socket.emit('sfx', 'buy');
    }
  });

  socket.on('updateDeck', (newDeck) => {
    const state = getState();
    if (newDeck.length <= CARDS_PER_SIDE && newDeck.every(id => state.collection.includes(id))) {
      state.deck = newDeck;
      socket.emit('stateUpdate', state);
    }
  });

  // ═══ БОЙ ═══
  socket.on('startBattle', () => {
    const state = getState();
    const playerCards = state.deck.map(id => {
      const c = getCard(id);
      return { ...c, maxHp: c.hp, currentHp: c.hp, mana: 2, critReady: false, tauntActive: false };
    });

    const availableEnemies = ALL_CARDS.filter(c => !state.deck.includes(c.id));
    const enemyCards = shuffle(availableEnemies).slice(0, CARDS_PER_SIDE).map(c => ({
      ...c, maxHp: c.hp, currentHp: c.hp, mana: 2, critReady: false, tauntActive: false
    }));

    state.battle = {
      playerCards,
      enemyCards,
      isPlayerTurn: true,
      activeIdx: -1,
      abilitiesUsed: new Set(),
      log: ['Бой начался. Тьма сгущается...'],
      gameOver: false,
      winner: null
    };
    
    // Преобразуем Set в массив для сериализации
    state.battle.abilitiesUsed = []; 
    socket.emit('stateUpdate', state);
  });

  socket.on('playerAction', (action) => {
    const state = getState();
    const battle = state.battle;
    if (!battle || battle.gameOver || !battle.isPlayerTurn) return;

    if (action.type === 'select') {
      battle.activeIdx = battle.activeIdx === action.idx ? -1 : action.idx;
      socket.emit('stateUpdate', state);
      if (battle.activeIdx !== -1) socket.emit('sfx', 'select');
      return;
    }

    if (action.type === 'attack' && battle.activeIdx !== -1) {
      const attacker = battle.playerCards[battle.activeIdx];
      const target = battle.enemyCards[action.targetIdx];
      
      if (!attacker || !target || target.currentHp <= 0) return;
      
      // Проверка провокации
      const taunter = battle.enemyCards.find(c => c.tauntActive && c.currentHp > 0);
      if (taunter && target.id !== taunter.id) return;

      const isCrit = attacker.type === 'assa' && attacker.critReady;
      const dmg = isCrit ? attacker.atk * 2 : attacker.atk;
      
      if (isCrit) attacker.critReady = false;
      
      const wasAlive = target.currentHp > 0;
      target.currentHp = Math.max(0, target.currentHp - dmg);
      const isDead = target.currentHp <= 0 && wasAlive;

      let logMsg = `<span class="log-player">${attacker.name}</span> атакует <span class="log-enemy">${target.name}</span> — <span class="log-dmg">${dmg} урона</span>`;
      if (isCrit) logMsg += ' <span class="log-crit">[КРИТ!]</span>';
      if (isDead) logMsg += ' <span class="log-death">[УБИТ]</span>';
      
      battle.log.push(logMsg);
      battle.activeIdx = -1;
      
      // Отправляем событие для анимации на клиенте
      socket.emit('combatEvent', { type: 'attack', attackerIdx: battle.activeIdx, targetIdx: action.targetIdx, dmg, isCrit, isDead, isPlayer: true });
      socket.emit('sfx', isCrit ? 'crit' : 'attack');

      setTimeout(() => {
        if (checkWinCondition(state, socket)) return;
        endPlayerTurn(state, socket);
      }, 800);
    }

    // Обработка способностей (упрощенно для примера)
    if (action.type === 'ability') {
        // Логика способностей аналогична: проверка маны, применение эффекта, лог, emit combatEvent
        // ...
    }
  });

  socket.on('disconnect', () => {
    players.delete(socket.id);
  });
});

// ═══ ВСПОМОГАТЕЛЬНЫЕ ФУНКЦИИ СЕРВЕРА ═══
function generateShop(playerId) {
  const state = players.get(playerId);
  const available = ALL_CARDS.filter(c => !state.collection.includes(c.id));
  state.shop = shuffle(available).slice(0, 12);
}

function endPlayerTurn(state, socket) {
  const battle = state.battle;
  battle.isPlayerTurn = false;
  battle.abilitiesUsed = []; // Сброс способностей
  socket.emit('stateUpdate', state);
  
  setTimeout(() => runAiTurn(state, socket), 1000);
}

function runAiTurn(state, socket) {
  const battle = state.battle;
  if (battle.gameOver) return;

  // Упрощенный ИИ: случайная атака живого юнита
  const aliveEnemies = battle.enemyCards.filter(c => c.currentHp > 0);
  const alivePlayers = battle.playerCards.filter(c => c.currentHp > 0);
  
  if (aliveEnemies.length === 0 || alivePlayers.length === 0) return;

  const attacker = aliveEnemies[0]; // Берем первого живого для простоты
  const target = alivePlayers[Math.floor(Math.random() * alivePlayers.length)];
  
  const dmg = attacker.atk;
  const wasAlive = target.currentHp > 0;
  target.currentHp = Math.max(0, target.currentHp - dmg);
  const isDead = target.currentHp <= 0 && wasAlive;

  const attackerIdx = battle.enemyCards.indexOf(attacker);
  const targetIdx = state.battle.playerCards.indexOf(target);

  battle.log.push(`<span class="log-enemy">${attacker.name}</span> атакует <span class="log-player">${target.name}</span> — <span class="log-dmg">-${dmg} HP</span>`);
  
  socket.emit('combatEvent', { type: 'attack', attackerIdx, targetIdx, dmg, isCrit: false, isDead, isPlayer: false });
  socket.emit('sfx', 'attack');

  setTimeout(() => {
    if (checkWinCondition(state, socket)) return;
    
    // Восстановление маны
    battle.playerCards.forEach(c => { if(c.currentHp>0) c.mana = Math.min(MAX_MANA, c.mana + 2); });
    battle.enemyCards.forEach(c => { if(c.currentHp>0) c.mana = Math.min(MAX_MANA, c.mana + 2); });
    
    battle.isPlayerTurn = true;
    socket.emit('stateUpdate', state);
  }, 800);
}

function checkWinCondition(state, socket) {
  const battle = state.battle;
  const pAlive = battle.playerCards.some(c => c.currentHp > 0);
  const eAlive = battle.enemyCards.some(c => c.currentHp > 0);

  if (!pAlive || !eAlive) {
    battle.gameOver = true;
    battle.winner = pAlive ? 'player' : 'enemy';
    
    if (battle.winner === 'player') {
      const goldReward = 40 + Math.floor(Math.random() * 40);
      state.gold += goldReward;
      battle.log.push(`<span class="log-victory">Победа! +${goldReward} золота</span>`);
      socket.emit('sfx', 'victory');
    } else {
      const goldReward = 10 + Math.floor(Math.random() * 10);
      state.gold += goldReward;
      battle.log.push(`<span class="log-defeat">Поражение... +${goldReward} золота</span>`);
      socket.emit('sfx', 'defeat');
    }
    
    socket.emit('stateUpdate', state);
    return true;
  }
  return false;
}

🔑 Ключевые изменения и преимущества:
Единый источник истины: Переменные playerGold, hp, mana больше не хранятся в браузере. Игрок не может открыть консоль и написать playerGold = 9999, потому что сервер проверит транзакцию.
Событийная модель анимаций: Сервер отправляет combatEvent (кто, кого, сколько урона, крит/нет). Клиент получает это событие и только тогда запускает CSS-анимацию тряски, частицы и всплывающие цифры. Это гарантирует, что анимация всегда соответствует реальным расчетам сервера.
ИИ на сервере: Функция runAiTurn теперь выполняется на Node.js. Клиент просто ждет обновления состояния. Это предотвращает подмену логики противника.
Готовность к расширению: Массив players в server.js легко заменяется на запросы к базе данных (MongoDB, PostgreSQL) для сохранения прогресса между перезагрузками страницы.
🚀 Как запустить:
Убедитесь, что у вас установлен Node.js.
Откройте терминал в папке проекта.
Выполните npm install.
Выполните node server.js.
Откройте в браузере http://localhost:3000.
Примечание: Не забудьте скопировать ваши оригинальные папки с изображениями (img/cards/, img/game/) в папку public/, чтобы они корректно отображались. Массив ALL_CARDS в server.js нужно дополнить всеми 15 картами из вашего исходного кода.
